        function openDetails(company, year, industry, count, reason, newsUrl) {

            document.getElementById('m-company').innerHTML = company + ' <span>Profile</span>';

            document.getElementById('m-year').innerText = year;

            document.getElementById('m-industry').innerText = industry;

            document.getElementById('m-count').innerText = count + ' employees affected';

            document.getElementById('m-reason').innerText = reason;

           

            // Map the link URL dynamically

            const linkBtn = document.getElementById('m-link');

            if (newsUrl && newsUrl !== '#') {

                linkBtn.href = newsUrl;

                linkBtn.style.display = 'inline-block';

            } else {

                linkBtn.style.display = 'none';

            }

           

            document.getElementById('detailsModal').classList.add('active');

        }

        function closeDetails() {

            document.getElementById('detailsModal').classList.remove('active');

        }

        function closeDetailsOutside(event) {

            if(event.target.id === 'detailsModal') {

                closeDetails();

            }

        }
